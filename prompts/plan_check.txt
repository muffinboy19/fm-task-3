Output ONLY valid JSON (no markdown fences, no commentary):
{
  "aligned": boolean,
  "confidence": "high" | "medium" | "low",
  "planned_files": ["path/from/plan.go"],
  "patch_files": ["path/from/diff.go"],
  "missing_from_patch": ["planned but not in diff"],
  "extra_in_patch": ["in diff but not in plan"],
  "deviations": ["short bullet explaining each mismatch"],
  "summary": "one sentence verdict"
}
