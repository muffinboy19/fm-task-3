+// Package jsonschema generates JSON Schema documents from struct validate tags.
+package jsonschema
+
+import (
+	"encoding/json"
+	"errors"
+	"fmt"
+	"reflect"
+)
+
+const defaultDraft = "https://json-schema.org/draft/2020-12/schema"
+
+// Option configures schema generation.
+type Option func(*config)
+
+type config struct {
+	useJSONTag           bool
+	tagName              string
+	requiredStructEnable bool
+	draft                string
+}
+
+// WithJSONTag uses json struct tag names for property keys (default: enabled).
+func WithJSONTag() Option {
+	return func(c *config) {
+		c.useJSONTag = true
+	}
+}
+
+// WithoutJSONTag uses Go field names for property keys.
+func WithoutJSONTag() Option {
+	return func(c *config) {
+		c.useJSONTag = false
+	}
+}
+
+// WithTagName sets the struct tag name used for validation rules (default: "validate").
+func WithTagName(name string) Option {
+	return func(c *config) {
+		c.tagName = name
+	}
+}
+
+// WithRequiredStructEnabled marks non-pointer struct fields with required as required in schema.
+func WithRequiredStructEnabled() Option {
+	return func(c *config) {
+		c.requiredStructEnable = true
+	}
+}
+
+// WithDraft sets the root $schema URI.
+func WithDraft(version string) Option {
+	return func(c *config) {
+		c.draft = version
+	}
+}
+
+func defaultConfig() *config {
+	return &config{
+		useJSONTag: true,
+		tagName:    "validate",
+		draft:      defaultDraft,
+	}
+}
+
+func applyOptions(opts ...Option) *config {
+	cfg := defaultConfig()
+	for _, opt := range opts {
+		opt(cfg)
+	}
+	if cfg.tagName == "" {
+		cfg.tagName = "validate"
+	}
+	if cfg.draft == "" {
+		cfg.draft = defaultDraft
+	}
+	return cfg
+}
+
+// GenerateSchema builds a JSON Schema document as a map for the given value v.
+// v must be a struct or a pointer to struct.
+func GenerateSchema(v any, opts ...Option) (map[string]any, error) {
+	cfg := applyOptions(opts...)
+	rt := reflect.TypeOf(v)
+	if rt == nil {
+		return nil, errors.New("jsonschema: nil input")
+	}
+	for rt.Kind() == reflect.Ptr {
+		rt = rt.Elem()
+	}
+	if rt.Kind() != reflect.Struct {
+		return nil, fmt.Errorf("jsonschema: expected struct, got %s", rt.Kind())
+	}
+	gen := &generator{cfg: cfg}
+	schema := gen.schemaForType(rt, parseTags("", cfg.tagName))
+	schema["$schema"] = cfg.draft
+	if _, ok := schema["type"]; !ok {
+		schema["type"] = "object"
+	}
+	return schema, nil
+}
+
+// Generate returns marshaled JSON Schema for v.
+func Generate(v any, opts ...Option) ([]byte, error) {
+	m, err := GenerateSchema(v, opts...)
+	if err != nil {
+		return nil, err
+	}
+	return json.Marshal(m)
+}

+	var m map[string]any

+	return m

+func prop(m map[string]any, name string) map[string]any {
+	props, _ := m["properties"].(map[string]any)
+	p, _ := props[name].(map[string]any)
+	return p
+}

+func requiredList(m map[string]any) []string {
+	r, _ := m["required"].([]any)
+	out := make([]string, len(r))
+	for i, v := range r {
+		out[i], _ = v.(string)
+	}
+	return out
+}

+func hasRequired(m map[string]any, name string) bool {
+	for _, r := range requiredList(m) {
+		if r == name {
+			return true
+		}
+	}
+	return false
+}

+package jsonschema
+
+import (
+	"reflect"
+)
+
+type generator struct {
+	cfg *config
+}
+
+func (g *generator) schemaForType(rt reflect.Type, tags *fieldTags) map[string]any {
+	for rt.Kind() == reflect.Ptr {
+		rt = rt.Elem()
+	}
+	schema := g.typeSchema(rt, tags)
+	if tags != nil && tags.afterDive != nil {
+		inner := g.schemaForType(rt, tags.afterDive)
+		switch rt.Kind() {
+		case reflect.Struct:
+			if props, ok := inner["properties"]; ok {
+				schema["properties"] = props
+			}
+			if req, ok := inner["required"]; ok {
+				schema["required"] = req
+			}
+		case reflect.Slice, reflect.Array:
+			schema["items"] = inner
+		case reflect.Map:
+			schema["additionalProperties"] = inner
+		}
+	}
+	return schema
+}
+
+func (g *generator) typeSchema(rt reflect.Type, tags *fieldTags) map[string]any {
+	schema := make(map[string]any)
+	if tags != nil && tags.booleanType {
+		schema["type"] = "boolean"
+		return schema
+	}
+	switch rt.Kind() {
+	case reflect.String:
+		schema["type"] = "string"
+		g.applyStringConstraints(schema, tags)
+	case reflect.Bool:
+		schema["type"] = "boolean"
+	case reflect.Int, reflect.Int8, reflect.Int16, reflect.Int32, reflect.Int64,
+		reflect.Uint, reflect.Uint8, reflect.Uint16, reflect.Uint32, reflect.Uint64:
+		schema["type"] = "integer"
+		g.applyNumericConstraints(schema, tags, true)
+	case reflect.Float32, reflect.Float64:
+		schema["type"] = "number"
+		g.applyNumericConstraints(schema, tags, false)
+	case reflect.Slice, reflect.Array:
+		schema["type"] = "array"
+		elem := rt.Elem()
+		itemTags := &fieldTags{}
+		if tags != nil && tags.afterDive != nil {
+			itemTags = tags.afterDive
+		}
+		schema["items"] = g.schemaForType(elem, itemTags)
+		g.applyArrayConstraints(schema, tags)
+	case reflect.Map:
+		schema["type"] = "object"
+		valTags := &fieldTags{}
+		if tags != nil && tags.afterDive != nil {
+			valTags = tags.afterDive
+		}
+		schema["additionalProperties"] = g.schemaForType(rt.Elem(), valTags)
+	case reflect.Struct:
+		g.buildStructSchema(rt, schema, tags)
+	default:
+		schema["type"] = "string"
+	}
+	if tags != nil {
+		g.applyCommonConstraints(schema, tags)
+	}
+	return schema
+}
+
+func (g *generator) buildStructSchema(rt reflect.Type, schema map[string]any, tags *fieldTags) {
+	schema["type"] = "object"
+	props := make(map[string]any)
+	var required []string
+	for i := 0; i < rt.NumField(); i++ {
+		fld := rt.Field(i)
+		if fld.Anonymous {
+			embedded := fld.Type
+			for embedded.Kind() == reflect.Ptr {
+				embedded = embedded.Elem()
+			}
+			if embedded.Kind() == reflect.Struct {
+				sub := g.schemaForStructFields(embedded)
+				for k, v := range sub.props {
+					props[k] = v
+				}
+				required = append(required, sub.required...)
+			}
+			continue
+		}
+		if len(fld.PkgPath) > 0 {
+			continue
+		}
+		vtag := fld.Tag.Get(g.cfg.tagName)
+		ft := parseTags(vtag, g.cfg.tagName)
+		if ft.skip {
+			continue
+		}
+		name := g.fieldName(fld)
+		if name == "" || name == "-" {
+			continue
+		}
+		ftags := ft
+		if ft.dive && ft.afterDive != nil {
+			ftags = mergeTags(ft, ft.afterDive)
+		}
+		fieldSchema := g.fieldSchema(fld.Type, ft, ftags)
+		props[name] = fieldSchema
+		if g.fieldRequired(fld.Type, ft) {
+			required = append(required, name)
+		}
+	}
+	schema["properties"] = props
+	if len(required) > 0 {
+		schema["required"] = required
+	}
+}
+
+type structFieldsResult struct {
+	props    map[string]any
+	required []string
+}
+
+func (g *generator) schemaForStructFields(rt reflect.Type) structFieldsResult {
+	s := make(map[string]any)
+	g.buildStructSchema(rt, s, nil)
+	props, _ := s["properties"].(map[string]any)
+	req, _ := s["required"].([]string)
+	return structFieldsResult{props: props, required: req}
+}
+
+func (g *generator) fieldSchema(rt reflect.Type, ft *fieldTags, constraintTags *fieldTags) map[string]any {
+	isPtr := rt.Kind() == reflect.Ptr
+	elem := rt
+	for elem.Kind() == reflect.Ptr {
+		elem = elem.Elem()
+	}
+	innerTags := ft
+	if ft.dive && ft.afterDive != nil {
+		innerTags = ft.afterDive
+	} else if !ft.dive {
+		innerTags = constraintTags
+	}
+	schema := g.schemaForType(elem, innerTags)
+	if isPtr && !g.fieldRequired(rt, ft) {
+		schema["nullable"] = true
+	}
+	return schema
+}
+
+func (g *generator) fieldRequired(rt reflect.Type, ft *fieldTags) bool {
+	if ft.omitEmpty || ft.omitNil || ft.omitZero {
+		return false
+	}
+	if !ft.required {
+		return false
+	}
+	elem := rt
+	for elem.Kind() == reflect.Ptr {
+		elem = elem.Elem()
+	}
+	if elem.Kind() == reflect.Struct {
+		return g.cfg.requiredStructEnable
+	}
+	return true
+}
+
+func (g *generator) fieldName(fld reflect.StructField) string {
+	if g.cfg.useJSONTag {
+		jt := fld.Tag.Get("json")
+		if jt != "" {
+			name := jt
+			if idx := indexComma(jt); idx >= 0 {
+				name = jt[:idx]
+			}
+			if name == "-" {
+				return ""
+			}
+			return name
+		}
+	}
+	return fld.Name
+}
+
+func indexComma(s string) int {
+	for i := 0; i < len(s); i++ {
+		if s[i] == ',' {
+			return i
+		}
+	}
+	return -1
+}
+
+func (g *generator) applyCommonConstraints(schema map[string]any, tags *fieldTags) {
+	if tags == nil {
+		return
+	}
+	if len(tags.oneof) > 0 {
+		enums := make([]any, len(tags.oneof))
+		for i, v := range tags.oneof {
+			enums[i] = v
+		}
+		schema["enum"] = enums
+	}
+	if tags.format != "" {
+		schema["format"] = tags.format
+	}
+	if tags.uniqueItems {
+		schema["uniqueItems"] = true
+	}
+}
+
+func (g *generator) applyStringConstraints(schema map[string]any, tags *fieldTags) {
+	if tags == nil {
+		return
+	}
+	if tags.minLen != nil {
+		schema["minLength"] = *tags.minLen
+	}
+	if tags.maxLen != nil {
+		schema["maxLength"] = *tags.maxLen
+	}
+	if tags.min != nil {
+		schema["minLength"] = int(*tags.min)
+	}
+	if tags.max != nil {
+		schema["maxLength"] = int(*tags.max)
+	}
+	if tags.eqStr != "" {
+		schema["minLength"] = len(tags.eqStr)
+		schema["maxLength"] = len(tags.eqStr)
+	}
+}
+
+func (g *generator) applyNumericConstraints(schema map[string]any, tags *fieldTags, integer bool) {
+	if tags == nil {
+		return
+	}
+	setNum := func(key string, v float64) {
+		if integer {
+			schema[key] = int(v)
+		} else {
+			schema[key] = v
+		}
+	}
+	if tags.eq != nil {
+		setNum("minimum", *tags.eq)
+		setNum("maximum", *tags.eq)
+	}
+	if tags.min != nil {
+		setNum("minimum", *tags.min)
+	}
+	if tags.max != nil {
+		setNum("maximum", *tags.max)
+	}
+	if tags.gt != nil {
+		schema["exclusiveMinimum"] = *tags.gt
+		if integer {
+			schema["exclusiveMinimum"] = int(*tags.gt)
+		}
+	}
+	if tags.gte != nil {
+		setNum("minimum", *tags.gte)
+	}
+	if tags.lt != nil {
+		schema["exclusiveMaximum"] = *tags.lt
+		if integer {
+			schema["exclusiveMaximum"] = int(*tags.lt)
+		}
+	}
+	if tags.lte != nil {
+		setNum("maximum", *tags.lte)
+	}
+}
+
+func (g *generator) applyArrayConstraints(schema map[string]any, tags *fieldTags) {
+	if tags == nil {
+		return
+	}
+	if tags.minLen != nil {
+		schema["minItems"] = *tags.minLen
+	}
+	if tags.maxLen != nil {
+		schema["maxItems"] = *tags.maxLen
+	}
+	if tags.min != nil {
+		schema["minItems"] = int(*tags.min)
+	}
+	if tags.max != nil {
+		schema["maxItems"] = int(*tags.max)
+	}
+	if tags.gt != nil {
+		n := int(*tags.gt)
+		schema["minItems"] = n + 1
+	}
+	if tags.gte != nil {
+		schema["minItems"] = int(*tags.gte)
+	}
+	if tags.lt != nil {
+		schema["maxItems"] = int(*tags.lt) - 1
+	}
+	if tags.lte != nil {
+		schema["maxItems"] = int(*tags.lte)
+	}
+	if tags.eq != nil {
+		n := int(*tags.eq)
+		schema["minItems"] = n
+		schema["maxItems"] = n
+	}

+	min            *float64
+	max            *float64
+	minLen         *int
+	maxLen         *int
+	eq             *float64
+	eqStr          string
+	gt             *float64
+	gte            *float64
+	lt             *float64
+	lte            *float64

+	booleanType    bool

+func parseTags(tag string, tagName string) *fieldTags {
+	if tag == "" {
+		return &fieldTags{}
+	}
+	if tag == skipTag {
+		return &fieldTags{skip: true}
+	}
+	return parseTagsRecursive(tag)
+}
+
+func parseTagsRecursive(tag string) *fieldTags {
+	ft := &fieldTags{}
+	tags := strings.Split(tag, tagSeparator)
+	var diveIdx int = -1
+	for i := 0; i < len(tags); i++ {
+		t := strings.TrimSpace(tags[i])
+		switch t {
+		case skipTag:
+			ft.skip = true
+			return ft
+		case diveTag:
+			ft.dive = true
+			diveIdx = i
+		case keysTag:
+			ft.keys = true
+			for i+1 < len(tags) && tags[i+1] != endKeysTag {
+				i++
+			}
+		case endKeysTag:
+		case "omitempty":
+			ft.omitEmpty = true
+		case "omitnil":
+			ft.omitNil = true
+		case "omitzero":
+			ft.omitZero = true
+		case "required":
+			ft.required = true
+		case "unique":
+			ft.uniqueItems = true
+		case "boolean":
+			ft.booleanType = true
+		case "email", "url", "uri", "uuid", "uuid4", "ipv4", "ipv6":
+			ft.format = t
+		default:
+			applyParamTag(ft, t)
+		}
+	}
+	if diveIdx >= 0 && diveIdx+1 < len(tags) {
+		rest := strings.Join(tags[diveIdx+1:], tagSeparator)
+		// strip keys/endkeys block from dive tail
+		if idx := strings.Index(rest, keysTag); idx >= 0 {
+			if end := strings.Index(rest, endKeysTag); end >= 0 {
+				after := rest[end+len(endKeysTag):]
+				rest = strings.TrimPrefix(after, tagSeparator)
+			}
+		}
+		rest = strings.TrimPrefix(rest, tagSeparator)
+		if rest != "" && rest != keysTag {
+			ft.afterDive = parseTagsRecursive(rest)
+		} else {
+			ft.afterDive = &fieldTags{}
+		}
+	}
+	return ft
+}
+
+func applyParamTag(ft *fieldTags, t string) {
+	parts := strings.SplitN(t, tagKeySeparator, 2)
+	key := parts[0]
+	param := ""
+	if len(parts) > 1 {
+		param = parts[1]
+	}
+	switch key {
+	case "min", "max", "eq", "gt", "gte", "lt", "lte":
+		if f, err := strconv.ParseFloat(param, 64); err == nil {
+			switch key {
+			case "min":
+				ft.min = &f
+			case "max":
+				ft.max = &f
+			case "eq":
+				ft.eq = &f
+			case "gt":
+				ft.gt = &f
+			case "gte":
+				ft.gte = &f
+			case "lt":
+				ft.lt = &f
+			case "lte":
+				ft.lte = &f
+			}
+		}
+	case "len":
+		if n, err := strconv.Atoi(param); err == nil {
+			switch {
+			case ft.minLen == nil:
+				ft.minLen = &n
+				ft.maxLen = &n
+			default:
+				ft.minLen = &n
+				ft.maxLen = &n
+			}
+		}
+	case "oneof":
+		ft.oneof = strings.Fields(param)
+	default:
+		// cross-field and unknown tags ignored in v1
+	}
+	if key == "eq" && param != "" && ft.eq == nil {
+		ft.eqStr = param
+	}
+}

+func mergeTags(parent, child *fieldTags) *fieldTags {
+	if child == nil {
+		return parent
+	}
+	out := *parent
+	if child.required {
+		out.required = true
+	}
+	if child.omitEmpty {
+		out.omitEmpty = true
+	}
+	if child.omitNil {
+		out.omitNil = true
+	}
+	if child.omitZero {
+		out.omitZero = true
+	}
+	if child.min != nil {
+		out.min = child.min
+	}
+	if child.max != nil {
+		out.max = child.max
+	}
+	if child.minLen != nil {
+		out.minLen = child.minLen
+	}
+	if child.maxLen != nil {
+		out.maxLen = child.maxLen
+	}
+	if child.eq != nil {
+		out.eq = child.eq
+	}
+	if child.eqStr != "" {
+		out.eqStr = child.eqStr
+	}
+	if child.gt != nil {
+		out.gt = child.gt
+	}
+	if child.gte != nil {
+		out.gte = child.gte
+	}
+	if child.lt != nil {
+		out.lt = child.lt
+	}
+	if child.lte != nil {
+		out.lte = child.lte
+	}
+	if len(child.oneof) > 0 {
+		out.oneof = child.oneof
+	}
+	if child.format != "" {
+		out.format = child.format
+	}
+	if child.uniqueItems {
+		out.uniqueItems = true
+	}
+	if child.booleanType {
+		out.booleanType = true
+	}
+	return &out
+}
